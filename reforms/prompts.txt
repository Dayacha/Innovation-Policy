def build_innovation_guidance(taxonomy_path=None):
    """Return a concise include/exclude keyword block built from search_library.json.

    Used to enrich the LLM system prompt so its understanding of 'innovation'
    matches the project's taxonomy exactly.
    """
    import json, pathlib

    if taxonomy_path is None:
        here = pathlib.Path(__file__).resolve().parent.parent
        taxonomy_path = here / "Data" / "input" / "taxonomy" / "search_library.json"

    try:
        with open(taxonomy_path) as f:
            lib = json.load(f)
    except (FileNotFoundError, json.JSONDecodeError):
        return ""   # graceful degradation — guidance is optional

    # Representative keywords per category (top 12 most distinctive each)
    core_rd   = lib["auto_include"]["keywords"][:20]
    inst      = lib["institutions"]["keywords"][:12]
    sectoral  = lib["sectoral_rd"]["keywords"][:12]
    budget    = lib["budget_terms"]["keywords"][:10]
    exclusions = lib["exclusions"]["keywords"][:20]

    # Ambiguous terms with their required anchors
    ambig_lines = []
    for term, rules in lib["ambiguous"]["terms"].items() if isinstance(lib["ambiguous"]["terms"], dict) else []:
        anchors = ", ".join(rules.get("require_anchor", [])[:4])
        ambig_lines.append(f'  · "{term}" — only if near: {anchors}')

    lines = [
        "── INNOVATION TAXONOMY (from project search library) ──────────────────",
        "",
        "INCLUDE: a reform is innovation-relevant if the text contains at least one of:",
        f"  Core R&D: {', '.join(core_rd[:15])}",
        f"  Institutions: {', '.join(inst[:10])}",
        f"  Sectoral R&D: {', '.join(sectoral[:10])}",
        f"  Budget instruments (combined with above): {', '.join(budget[:8])}",
        "",
        "AMBIGUOUS — include only if a required anchor is nearby:",
    ] + ambig_lines + [
        "",
        "EXCLUDE — these are false positives, do not classify as innovation:",
        f"  {', '.join(exclusions)}",
        "",
        "────────────────────────────────────────────────────────────────────────",
    ]
    return "\n".join(lines)


def _build_theme_block():
    """Build the theme/subtheme definition block for prompts.

    For each theme we emit the theme key, label and description, then for
    every sub-theme the key, short description and cue keywords.  This
    richer format helps the LLM distinguish adjacent themes and pick the
    right sub-theme.
    """
    lines = []
    for theme_key, info in THEMES_SUBTHEMES.items():
        lines.append(
            f'- "{theme_key}" ({info["label"]}): {info["description"]}'
        )
        lines.append("  Sub-themes:")
        for st_key, st_info in info["subthemes"].items():
            kw = ", ".join(st_info["cue_keywords"]) if st_info["cue_keywords"] else "—"
            lines.append(
                f'    · "{st_key}" — {st_info["description"]}'
                f'\n      Keywords: {kw}'
            )
        lines.append("")   # blank line between themes
    return "\n".join(lines)


THEME_DEFINITIONS_BLOCK = _build_theme_block()


def build_innovation_system_prompt(taxonomy_path=None):
    """Build the specialised system prompt for innovation-only extraction.

    Loads the project taxonomy from search_library.json and embeds the
    include/exclude rules directly into the system prompt so the LLM
    applies the same criteria as the budget pipeline (Stream 1).
    """
    # Build the innovation subtheme block (only the innovation theme)
    innovation_info = THEMES_SUBTHEMES["innovation"]
    st_lines = []
    for st_key, st_info in innovation_info["subthemes"].items():
        kw = ", ".join(st_info["cue_keywords"]) if st_info["cue_keywords"] else "—"
        st_lines.append(
            f'  · "{st_key}" ({st_info["label"]})\n'
            f'    {st_info["description"]}\n'
            f'    Keywords: {kw}'
        )
    subtheme_block = "\n\n".join(st_lines)

    rd_actor_block = "\n".join(
        f'  · "{k}": {v}' for k, v in RD_ACTORS.items()
    )
    rd_stage_block = "\n".join(
        f'  · "{k}": {v}' for k, v in RD_STAGES.items()
    )

    taxonomy_guidance = build_innovation_guidance(taxonomy_path)

    return f"""\
You are an expert political economist specialising in science, technology \
and innovation (STI) policy. Your task is to extract and \
classify innovation policy reforms from text.

You must return your analysis as valid JSON only — no commentary, no markdown \
fences, no explanation outside the JSON structure.

── SCOPE ───────────────────────────────────────────────────────────────────────
Extract ONLY reforms that directly relate to innovation policy: R&D funding, \
innovation instruments, research infrastructure, knowledge transfer, startup \
ecosystems, human capital for research, or sectoral/mission R&D.

A reform must represent a significant policy change — not a minor administrative \
adjustment or a general statement about innovation ambitions.

{taxonomy_guidance}

── INNOVATION TYPE (sub_theme) ─────────────────────────────────────────────────
Classify every reform into exactly ONE of these eight types:

{subtheme_block}

Rules for "other":
- Use ONLY when the reform is unambiguously innovation-relevant but truly \
does not fit any of the seven types above.
- You must explain why in the description field.
- If in doubt between two types, choose the more specific one.

── R&D ACTOR (rd_actor) ────────────────────────────────────────────────────────
Who is the primary target or beneficiary of the reform?

{rd_actor_block}

── R&D STAGE (rd_stage) ────────────────────────────────────────────────────────
Where in the research-to-market pipeline does this reform intervene?

{rd_stage_block}

── SECONDARY TYPE (secondary_type) ─────────────────────────────────────────────
If a reform GENUINELY AND SUBSTANTIALLY spans a second innovation type, provide \
it in secondary_type (use the same eight values as sub_theme). Set to null if \
the reform clearly belongs to one type. Examples:
- A sectoral health R&D fund that also builds a new lab → \
sub_theme="sectoral_rd", secondary_type="research_infrastructure"
- A doctoral fellowship programme tied to an industry collaboration scheme → \
sub_theme="human_capital", secondary_type="knowledge_transfer"

── GROWTH ORIENTATION ───────────────────────────────────────────────────────────
Classify the expected medium- to long-run effect on GDP per capita:
- "growth_supporting": raises productivity, innovation, human capital, or \
  allocative efficiency
- "growth_hindering": reduces incentives or creates distortions
- "mixed": meaningful trade-offs in both directions
- "unclear_or_neutral": genuinely ambiguous or minor

── STATUS ───────────────────────────────────────────────────────────────────────
- "implemented": the reform is in force
- "legislated": passed into law but not yet implemented
- "announced": officially announced but not yet legislated

── IMPORTANCE ───────────────────────────────────────────────────────────────────
- is_major_reform=true / importance_bucket=3: structural, system-wide change \
  (e.g., creation of a national research agency, major overhaul of R&D tax credits)
- importance_bucket=2: significant but narrower (e.g., new sectoral fund, \
  extension of an existing scheme)
- importance_bucket=1: minor or incremental adjustment
"""

SYSTEM_PROMPT = """\
You are an expert political economist specializing in structural economic \
reforms. Your task is to extract and classify economic \
reforms discussed in Economic Surveys.

You must return your analysis as valid JSON only — no commentary, no markdown \
fences, no explanation outside the JSON structure.

Key definitions:
- A "reform" is a significant policy change that alters the regulatory, \
institutional, or fiscal framework of the economy. Minor administrative \
adjustments do not count.

THEMES AND SUBTHEMES:
Each reform must be classified into the theme(s) listed below, \
and assigned one of the listed sub_themes (use "other" only as a last resort). \
The cue keywords are illustrative, not exhaustive.

""" + THEME_DEFINITIONS_BLOCK + """

THEME CLASSIFICATION GUIDANCE:
- "theme": Always provide the single best-fitting primary theme.
- "sub_theme": Always provide the single best-fitting sub-theme \
within the chosen theme. Avoid "other".
- "secondary_type": Use this when the reform GENUINELY AND \
SUBSTANTIALLY spans a second innovation sub-type (e.g., a sectoral \
health R&D fund that also builds a new lab). Leave null otherwise.
- "alternative_theme": Use this when you are UNCERTAIN about the \
primary sub_theme assignment — provide the most plausible alternative \
here. Leave null if you are confident. \
Do NOT set both secondary_type and alternative_theme.

PACKAGE vs COMPONENT distinction:
- A "reform package" is a major legislative or policy initiative that may \
contain multiple components (e.g., "2023 Pension Reform" is a package; \
"raise statutory retirement age from 62 to 64" and "abolish special \
retirement regimes" are components of that package).
- When a reform is part of a broader package, set is_component=true and \
provide a package_name that would be shared across all components of \
the same package.
- When a reform stands alone, set is_component=false and leave \
package_name as the reform's own canonical name.

For GROWTH ORIENTATION — classify each reform according to its expected \
medium- to long-run effect on GDP per capita, in the spirit of \
structural policy analysis (Going for Growth / Foundations for Growth).

Important framing rules:
- This is a classification of the reform's expected structural orientation, \
not a claim about its realised causal effect in practice.
- Base the label on the reform description and quoted evidence provided.
- Focus on likely effects through productivity, employment and \
labour-force participation, capital deepening, innovation, allocative \
efficiency, competition, skill formation, resilience, and the reduction of \
persistent distortions or bottlenecks.
- If the description is too vague or the growth implications are genuinely \
ambiguous, use "unclear_or_neutral".
- Do not infer a strong growth orientation unless the description supports it.
- Minor reforms with a clear growth direction should not be classified as \
"unclear_or_neutral" just because they are minor.

Allowed values for "growth_orientation":
- "growth_supporting": The reform is expected to raise medium- to long-run \
GDP per capita by improving productivity, labour utilisation, investment \
incentives, innovation, competition, resource allocation, human capital, or \
the efficiency/resilience of key economic systems.
- "growth_hindering": The reform is expected to reduce medium- to long-run \
GDP per capita by increasing distortions, weakening competition, reducing \
investment or innovation incentives, lowering labour mobility/participation, \
worsening resource allocation, or creating persistent inefficiencies/barriers.
- "mixed": The reform contains both growth-supporting and growth-hindering \
elements, or involves material trade-offs such that no single direction \
clearly dominates from the description alone.
- "unclear_or_neutral": The reform is mainly administrative, institutional, \
distributive, or social in nature with no clear growth direction from the \
description, or the text does not provide enough information to classify \
it confidently.

Decision rules:
1. Classify based on the expected structural effect on growth, not on \
political desirability or social equity.
2. Use the reform's likely medium- to long-run economic mechanism, \
not short-run demand effects.
3. If a reform has clear trade-offs and neither side clearly dominates, \
choose "mixed".
4. If the reform is primarily about redistribution, governance, coverage, \
fairness, or institutional organisation and the growth channel is weak or \
unclear in the text, choose "unclear_or_neutral".
5. If the reform is a package, classify the package as a whole; if the \
mention is only one component, classify that component.
6. Be conservative: when uncertain, prefer "unclear_or_neutral" over an \
overconfident directional label.

Theme-specific guidance for innovation reforms:
- innovation / rd_funding: growth_supporting if it increases the quantity \
or quality of public R&D funding, improves allocation efficiency, or \
strengthens research councils and universities; growth_hindering if it cuts \
funding without strategic rationale or distorts allocation.
- innovation / innovation_instruments: growth_supporting if it introduces \
or improves R&D tax credits, direct grants, or innovation governance in \
ways that raise private R&D incentives; growth_hindering if it creates \
poorly targeted subsidies or bureaucratic duplication.
- innovation / research_infrastructure: growth_supporting if it expands \
shared labs, science parks, or data infrastructure that raise research \
productivity; "unclear_or_neutral" if it is mainly administrative.
- innovation / knowledge_transfer: growth_supporting if it reduces \
university-industry friction, strengthens TTOs, or improves IP regimes; \
growth_hindering if it creates IP barriers or weakens open science.
- innovation / startup_ecosystem: growth_supporting if it lowers entry \
costs for innovative firms or expands early-stage financing; growth_hindering \
if it creates poorly targeted support or displaces private capital.
- innovation / human_capital: growth_supporting if it expands the supply \
of researchers or improves mobility and career incentives; growth_hindering \
if it reduces research career attractiveness or worsens brain drain.
- innovation / sectoral_rd: growth_supporting if it directs R&D toward \
areas of high social return (health, climate, digital); "mixed" if it \
crowds out basic research or introduces mission-specific distortions.
- innovation / other: apply the general growth_supporting / growth_hindering \
logic described above, noting the specific innovation channel in the rationale.

For STATUS classification:
- "implemented": The reform has begun to be implemented, applied, enforced, \
rolled out, or put into effect/force by the responsible administrative body \

service, or sector-specific body). Evidence: past-tense language about the \
reform being in effect ("came into force", "was implemented", "has been \
operational since", "began to be applied").
- "legislated": The reform has been formally enacted, passed, approved, \
ratified, promulgated, adopted or codified into law, via legislative or \
